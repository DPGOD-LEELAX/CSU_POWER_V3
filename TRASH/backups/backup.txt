    const loadDiff = loadMultiplier - 1.0;
    const pfDiff = 0.95 - powerFactor;
    
    for (let i = 0; i < 7; i++) {
        if (systemType === 'distributed') {
            // Start with nominal
            voltages[i] = distNominal[i];
            
            // Load effect
            if (loadDiff > 0) {
                voltages[i] -= loadDiff * (0.04 + (i * 0.01));
            } else if (loadDiff < 0) {
                voltages[i] += Math.abs(loadDiff) * (0.03 - (i * 0.002));
            }
            
            // PF effect
            voltages[i] -= pfDiff * (0.03 + (i * 0.005));
            
        } else {
            // CENTRALIZED
            voltages[i] = centNominal[i];
            
            // Load effect - stronger
            if (loadDiff > 0) {
                voltages[i] -= loadDiff * (0.06 + (i * 0.015));
            } else if (loadDiff < 0) {
                voltages[i] += Math.abs(loadDiff) * (0.02 - (i * 0.001));
            }
            
            // PF effect - stronger
            voltages[i] -= pfDiff * (0.05 + (i * 0.008));
        }
    }
    
    // Apply bounds - NO ARTIFICIAL FLOOR
    const minVoltage = 0.75;
    const maxVoltage = 1.02;
    
    for (let i = 0; i < 7; i++) {
        voltages[i] = Math.max(minVoltage, Math.min(maxVoltage, voltages[i]));
    }
    
    return voltages;
}

        function getVoltageColor(voltage) {
            if (voltage >= 0.97) return '#00ff00';
            if (voltage >= 0.95) return '#ffff00';
            if (voltage >= 0.93) return '#ff8000';
            return '#ff0000';
        }

                peakDemand: peakFromHistorical,
                avgDemand: totalLoad,
                peakTime: '14:00 - 16:00'
            };
        }

        function updateComplianceDisplay() {
            busVoltages = calculateVoltages();
            const worstV = Math.min(...busVoltages);
            const worstIdx = busVoltages.indexOf(worstV);
            const dropPercent = (1.0 - worstV) * 100;
            const isCompliant = worstV >= 0.95;
            
            let html = `
                <span style="color: #ffd700;">⚡ PEC 2025 VOLTAGE DROP ANALYSIS</span><br>
                <span style="color: #ffd700;">──────────────────────────────</span><br>

            `;
            
            if (isCompliant) {
                html += `<span style="color: #27ae60; font-size: 1.2em;">✓ PEC COMPLIANT</span><br>`;
            } else {
                html += `<span style="color: #e74c3c; font-size: 1.2em;">❌ PEC VIOLATION</span><br>`;
            }
            
            html += `<br><span style="color: #ffd700;">Maximum Allowable Drop:</span> 5.00%<br>`;
            html += `<span style="color: #ffd700;">Current Worst Drop:</span> <span style="color: ${isCompliant ? '#27ae60' : '#e74c3c'};">${dropPercent.toFixed(2)}%</span><br>`;

            if (worstV < 0.93) {
                html += `<span style="color: #e74c3c;">🔴 CRITICAL - Immediate action required</span><br>`;
                html += `• Install capacitor bank at Bus ${worstIdx+1}<br>`;
                html += `• Consider load reduction<br>`;

                html += `<span style="color: #ffd700;">🟡 WARNING - Schedule voltage correction</span><br>`;

                html += `• Continue monitoring<br>`;
                html += `• Maintain power factor >0.95`;
            }
            
            document.getElementById('complianceDisplay').innerHTML = html;
        }
        
        // Drawing functions
        function drawLine(ctx, a, b, dashed = false, color = '#e67e22', width = 3, active = true) {
            ctx.beginPath();
            ctx.moveTo(a.x, a.y);
            ctx.lineTo(b.x, b.y);
            
            if (!active) {
                ctx.strokeStyle = '#aaaaaa';
                ctx.lineWidth = width * 0.7;
                ctx.setLineDash([5, 5]);
            } else {
                ctx.strokeStyle = color;
                ctx.lineWidth = width;
                if (dashed) {
                    ctx.setLineDash([8, 5]);
                } else {
                    ctx.setLineDash([]);
                }
            }
            
            ctx.stroke();
            ctx.setLineDash([]);
        }
        
        function drawCanvas() {
            const canvas = document.getElementById('powerCanvas');
            const ctx = canvas.getContext('2d');
            
            ctx.save();
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.translate(offsetX, offsetY);
            ctx.scale(scale, scale);
            
            // Draw background grid
            ctx.strokeStyle = '#e0e0e0';
            ctx.lineWidth = 1 / scale;
            
            for (let x = 0; x <= canvas.width; x += 100) {
                ctx.beginPath();
                ctx.strokeStyle = '#e0e0e0';
                ctx.moveTo(x, 0);
                ctx.lineTo(x, canvas.height);
                ctx.stroke();
            }
            
            for (let y = 0; y <= canvas.height; y += 100) {
                ctx.beginPath();
                ctx.strokeStyle = '#e0e0e0';
                ctx.moveTo(0, y);
                ctx.lineTo(canvas.width, y);
                ctx.stroke();
            }
            
            // Calculate latest voltages
            busVoltages = calculateVoltages();
            
            // Draw connections based on system type
            if (systemType === 'distributed') {
                drawLine(ctx, sourcePos, distTxPos[0], true, '#ff6600', 3, true);
                drawLine(ctx, sourcePos, distTxPos[1], true, '#ff6600', 3, true);
                drawLine(ctx, sourcePos, distTxPos[2], true, '#ff6600', 3, true);
                
                drawLine(ctx, distTxPos[0], meterPos['Meter A'], false, '#3366cc', 2.5, true);
                drawLine(ctx, distTxPos[1], meterPos['Meter B'], false, '#3366cc', 2.5, true);
                drawLine(ctx, distTxPos[2], meterPos['Meter C'], false, '#3366cc', 2.5, true);
                
                for (let i = 0; i <= 2; i++) {
                    drawLine(ctx, meterPos['Meter A'], busPositions[i], true, '#3366cc', 2, true);
                }
                for (let i = 3; i <= 4; i++) {
                    drawLine(ctx, meterPos['Meter B'], busPositions[i], true, '#3366cc', 2, true);
                }
                for (let i = 5; i <= 6; i++) {
                    drawLine(ctx, meterPos['Meter C'], busPositions[i], true, '#3366cc', 2, true);
                }
                
                drawLine(ctx, sourcePos, mainTxPos, true, '#aaaaaa', 2, false);
                
            } else {
                drawLine(ctx, sourcePos, mainTxPos, true, '#ff6600', 3, true);
                
                for (let i = 0; i < busPositions.length; i++) {
                    drawLine(ctx, mainTxPos, busPositions[i], false, '#3366cc', 2, true);
                }
                
                drawLine(ctx, sourcePos, distTxPos[0], true, '#aaaaaa', 2, false);
                drawLine(ctx, sourcePos, distTxPos[1], true, '#aaaaaa', 2, false);
                drawLine(ctx, sourcePos, distTxPos[2], true, '#aaaaaa', 2, false);
                
                drawLine(ctx, distTxPos[0], meterPos['Meter A'], false, '#aaaaaa', 2, false);
                drawLine(ctx, distTxPos[1], meterPos['Meter B'], false, '#aaaaaa', 2, false);
                drawLine(ctx, distTxPos[2], meterPos['Meter C'], false, '#aaaaaa', 2, false);
                
                for (let i = 0; i <= 2; i++) {
                    drawLine(ctx, meterPos['Meter A'], busPositions[i], true, '#aaaaaa', 2, false);
                }
                for (let i = 3; i <= 4; i++) {
                    drawLine(ctx, meterPos['Meter B'], busPositions[i], true, '#aaaaaa', 2, false);
                }
                for (let i = 5; i <= 6; i++) {
                    drawLine(ctx, meterPos['Meter C'], busPositions[i], true, '#aaaaaa', 2, false);
                }
            }
            
            // Draw animated power flow
            ctx.shadowColor = '#ffff00';
            ctx.shadowBlur = 10;
            
            particles.forEach(flow => {
                flow.progress += flow.speed * animationSpeed;
                if (flow.progress > 1.0) flow.progress = 0;
                
                const x = flow.start.x + (flow.end.x - flow.start.x) * flow.progress;
                const y = flow.start.y + (flow.end.y - flow.start.y) * flow.progress;
                
                ctx.beginPath();
                ctx.arc(x, y, 5, 0, Math.PI * 2);
                ctx.fillStyle = '#ffff00';
                ctx.fill();
                ctx.strokeStyle = '#ff9900';
                ctx.lineWidth = 1;
                ctx.stroke();
            });
            
            ctx.shadowBlur = 0;
            
            // Draw source
            ctx.beginPath();
            ctx.arc(sourcePos.x, sourcePos.y, 25, 0, Math.PI * 2);
            ctx.fillStyle = '#ffaa00';
            ctx.fill();
            ctx.strokeStyle = '#000000';
            ctx.lineWidth = 2;
            ctx.stroke();
            
            ctx.font = 'bold 10px Arial';
            ctx.fillStyle = '#000000';
            ctx.textAlign = 'center';
            ctx.fillText('CAGELCO', sourcePos.x, sourcePos.y - 20);
            ctx.font = '8px Arial';
            ctx.fillText('150 kV', sourcePos.x, sourcePos.y + 8);
            
            // Draw main transformer
            ctx.beginPath();
            ctx.ellipse(mainTxPos.x, mainTxPos.y, 20, 17, 0, 0, Math.PI * 2);
            ctx.fillStyle = systemType === 'centralized' ? '#ffcc99' : '#cccccc';
            ctx.fill();
            ctx.strokeStyle = '#000000';
            ctx.lineWidth = 2;
            ctx.stroke();
            
            ctx.beginPath();
            ctx.moveTo(mainTxPos.x - 12, mainTxPos.y - 8);
            ctx.lineTo(mainTxPos.x + 12, mainTxPos.y + 8);
            ctx.stroke();
            ctx.beginPath();
            ctx.moveTo(mainTxPos.x - 12, mainTxPos.y + 8);
            ctx.lineTo(mainTxPos.x + 12, mainTxPos.y - 8);
            ctx.stroke();
            
            ctx.font = 'bold 8px Arial';
            ctx.fillStyle = '#000000';
            ctx.fillText('Main Tr', mainTxPos.x, mainTxPos.y - 20);
            
            // Draw distribution transformers
            distTxPos.forEach((pos, idx) => {
                ctx.beginPath();
                ctx.ellipse(pos.x, pos.y, 18, 15, 0, 0, Math.PI * 2);
                
                if (selectedElement === pos && selectedType === 'distTr') {
                    ctx.fillStyle = '#ff9999';
                } else {
                    ctx.fillStyle = systemType === 'distributed' ? '#99ccff' : '#dddddd';
                }
                
                ctx.fill();
                ctx.strokeStyle = '#000';
                ctx.stroke();
                
                ctx.font = 'bold 7px Arial';
                ctx.fillStyle = '#000';
                ctx.fillText(`Dist Tr${idx+1}`, pos.x, pos.y - 15);
                
                if (systemType === 'centralized') {
                    ctx.font = '6px Arial';
                    ctx.fillStyle = '#666';
                    ctx.fillText('idle', pos.x, pos.y + 18);
                }
            });
            
            // Draw meters
            for (let [name, pos] of Object.entries(meterPos)) {
                ctx.fillStyle = systemType === 'distributed' ? '#f0f4fa' : '#eeeeee';
                ctx.strokeStyle = '#000';
                ctx.lineWidth = 1.5;
                ctx.fillRect(pos.x - 28, pos.y - 15, 56, 30);
                ctx.strokeRect(pos.x - 28, pos.y - 15, 56, 30);
                
                ctx.font = 'bold 7px Arial';
                ctx.fillStyle = '#000';
                ctx.fillText(name, pos.x - 12, pos.y + 2);
                
                if (systemType === 'centralized') {
                    ctx.font = '5px Arial';
                    ctx.fillStyle = '#666';
                    ctx.fillText('idle', pos.x - 5, pos.y + 12);
                }
            }
            
            // Draw all poles/buses
            for (let i = 0; i < 7; i++) {
                const v = busVoltages[i];
                const color = getVoltageColor(v);
                const pos = busPositions[i];
                
                ctx.fillStyle = color;
                ctx.strokeStyle = '#000';
                ctx.lineWidth = 2;
                ctx.fillRect(pos.x - 35, pos.y - 25, 70, 50);
                ctx.strokeRect(pos.x - 35, pos.y - 25, 70, 50);
                
                ctx.font = 'bold 9px Arial';
                ctx.fillStyle = '#000';
                ctx.textAlign = 'center';
                
                let shortName = `Bus ${i+1}`;
                if (i === 0) shortName += ' (Ad)';
                else if (i === 1) shortName += ' (En)';
                else if (i === 2) shortName += ' (Li)';
                else if (i === 3) shortName += ' (Sc)';
                else if (i === 4) shortName += ' (Cl)';
                else if (i === 5) shortName += ' (Gy)';
                else shortName += ' (Do)';
                
                ctx.fillText(shortName, pos.x, pos.y - 10);
                
                ctx.font = 'bold 8px monospace';
                ctx.fillStyle = '#000';
                ctx.fillText(v.toFixed(3) + ' pu', pos.x, pos.y + 8);
                
                const drop = ((1.0 - v) * 100).toFixed(1);
                ctx.font = '7px Arial';
                ctx.fillStyle = '#333';
                ctx.fillText(drop + '% drop', pos.x, pos.y + 22);
                
                // Add span distance below pole
                ctx.font = '6px Arial';
                ctx.fillStyle = '#666';
                ctx.fillText(spanDistances[i] + 'm span', pos.x, pos.y + 35);
            }
            
            // Highlight selected element
            if (selectedElement && mode === 'select') {
                ctx.strokeStyle = '#ff0000';
                ctx.lineWidth = 3;
                ctx.setLineDash([5, 3]);
                
                if (selectedType === 'bus') {
                    ctx.strokeRect(
                        selectedElement.x - 40, 
                        selectedElement.y - 30, 
                        80, 60
                    );
                } else if (selectedType === 'meter') {
                    ctx.strokeRect(
                        selectedElement.x - 33, 
                        selectedElement.y - 20, 
                        66, 40
                    );
                } else if (selectedType === 'distTr') {
                    ctx.strokeRect(
                        selectedElement.x - 23, 
                        selectedElement.y - 20, 
                        46, 40
                    );
                }
                
                ctx.setLineDash([]);
            }
            
            ctx.restore();
            
            // Update displays
            updateLoadMetrics();
            updateDataDisplay();
            updateComplianceDisplay();
            
            if (animationRunning) {
                animationFrame = requestAnimationFrame(drawCanvas);
            }
        }

        function findElementAt(x, y) {
            for (let i = 0; i < busPositions.length; i++) {
                const bus = busPositions[i];
                if (Math.abs(x - bus.x) < 40 && Math.abs(y - bus.y) < 30) {
                    return { type: 'bus', element: bus, index: i };
                }
            }
            
            for (let i = 0; i < distTxPos.length; i++) {
                const tr = distTxPos[i];
                if (Math.abs(x - tr.x) < 25 && Math.abs(y - tr.y) < 20) {
                    return { type: 'distTr', element: tr, index: i };
                }
            }
            
            for (let [name, pos] of Object.entries(meterPos)) {
                if (Math.abs(x - pos.x) < 30 && Math.abs(y - pos.y) < 20) {
                    return { type: 'meter', element: pos, name: name };
                }
            }
            
            return null;
        }
        
        // Event listeners for canvas interaction
        document.addEventListener('DOMContentLoaded', function() {
            const container = document.getElementById('canvasContainer');
            const canvas = document.getElementById('powerCanvas');
            
            container.addEventListener('mousedown', function(e) {
                e.preventDefault();
                
                const rect = canvas.getBoundingClientRect();
                const mouseX = (e.clientX - rect.left - offsetX) / scale;
                const mouseY = (e.clientY - rect.top - offsetY) / scale;
                
                if (mode === 'select') {
                    const found = findElementAt(mouseX, mouseY);
                    if (found) {
                        selectedElement = found.element;
                        selectedType = found.type;
                        selectedIndex = found.index;
                        
                        dragOffsetX = mouseX - selectedElement.x;
                        dragOffsetY = mouseY - selectedElement.y;
                        
                        isDragging = true;
                        container.style.cursor = 'grabbing';
                    } else {
                        selectedElement = null;
                        selectedType = null;
                        selectedIndex = null;
                    }
                } else {
                    isDragging = true;
                    lastX = e.clientX;
                    lastY = e.clientY;
                    container.style.cursor = 'grabbing';
                }
            });
            
            container.addEventListener('mousemove', function(e) {
                if (!isDragging) return;
                
                const rect = canvas.getBoundingClientRect();
                
                if (mode === 'pan') {
                    const dx = e.clientX - lastX;
                    const dy = e.clientY - lastY;
                    
                    offsetX += dx;
                    offsetY += dy;
                    
                    lastX = e.clientX;
                    lastY = e.clientY;
                } else if (mode === 'select' && selectedElement) {
                    const mouseX = (e.clientX - rect.left - offsetX) / scale;
                    const mouseY = (e.clientY - rect.top - offsetY) / scale;
                    
                    selectedElement.x = mouseX - dragOffsetX;
                    selectedElement.y = mouseY - dragOffsetY;
                    
                    initParticles();
                }
                
                drawCanvas();
            });
            
            container.addEventListener('mouseup', function() {
                isDragging = false;
                container.style.cursor = mode === 'select' ? 'default' : 'grab';
            });
            
            container.addEventListener('mouseleave', function() {
                isDragging = false;
                container.style.cursor = mode === 'select' ? 'default' : 'grab';
            });
            
            container.addEventListener('wheel', function(e) {
                e.preventDefault();
                
                const rect = canvas.getBoundingClientRect();
                const mouseX = e.clientX - rect.left;
                const mouseY = e.clientY - rect.top;
                
                const worldX = (mouseX - offsetX) / scale;
                const worldY = (mouseY - offsetY) / scale;
                
                if (e.ctrlKey) {
                    const delta = e.deltaY > 0 ? 0.9 : 1.1;
                    scale = Math.max(0.3, Math.min(3.0, scale * delta));
                    
                    offsetX = mouseX - worldX * scale;
                    offsetY = mouseY - worldY * scale;
                    
                    document.getElementById('zoomLevel').textContent = Math.round(scale * 100) + '%';
                    drawCanvas();
                } else {
                    offsetX -= e.deltaX;
                    offsetY -= e.deltaY;
                    drawCanvas();
                }
            });
        });
